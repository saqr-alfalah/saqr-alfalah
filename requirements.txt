@app.route('/webhook', methods=['POST'])
def webhook():
    if request.args.get('secret')!= SECRET:
        return 'Unauthorized', 403
    update = telebot.types.Update.de_json(request.get_data().decode('utf-8'))
    bot.process_new_updates([update])
    return 'ok', 200

@bot.message_handler(commands=['start'])

@bot.message_handler(content_types=['photo'])
def handle_photo_gemini(message):
    msg = bot.reply_to(message, "جاري تحليل الصورة مع Gemini... ⏳")
    try:
        file_id = message.photo[-1].file_id
        file_info = bot.get_file(file_id)
        file_url = f"https://api.telegram.org/file/bot{TOKEN}/{file_info.file_path}"
        response = requests.get(file_url)
        img = Image.open(BytesIO(response.content))

        prompt = """انت خبير مشاريع مياه وبنية تحتية في السعودية تابع لشركة صقر الفلاح.
        حلل صورة المخالفة الميدانية هذي واعطني تقرير مختصر بالنقاط:
        1. نوع المخالفة: وصف دقيق للمشكلة
        2. مستوى الخطورة: منخفض/متوسط/عالي/كارثي
        3. التوصية الفنية العاجلة: وش يسوي المراقب الحين
        4. البند المخالف: حسب كود وزارة البيئة والمياه والزراعة
        5. التكلفة التقديرية للإصلاح
        رد بالعربي وبشكل مباشر للمراقب الميداني."""

        result = model.generate_content([prompt, img])
        ai_result = result.text

        final_report = f"🤖 تحليل Gemini المجاني:\n\n{ai_result}\n\nوقت البلاغ: {datetime.now().strftime('%Y-%m-%d %H:%M')}\nالمبلغ: @{message.from_user.username}"
        
        bot.edit_message_text(final_report, message.chat.id, msg.message_id)
        bot.send_message(ADMIN_CHAT_ID, f"🚨 بلاغ جديد:\n\n{final_report}")
        bot.send_photo(ADMIN_CHAT_ID, file_id)

    except Exception as e:
        bot.edit_message_text(f"صار خطأ: {str(e)}", message.chat.id, msg.message_id)

@bot.message_handler(func=lambda message: message.text == "📊 استشارة فنية")
def ask_question(message):
    bot.reply_to(message, "اكتب سؤالك الفني عن مشاريع المياه والبنية التحتية:")
    bot.register_next_step_handler(message, ai_consult_gemini)

def ai_consult_gemini(message):
    try:
        prompt = f"انت استشاري أنظمة مياه وبنية تحتية في السعودية. جاوب على هذا السؤال باختصار وبدقة فنية حسب كود الوزارة: {message.text}"
        response = model.generate_content(prompt)
        bot.reply_to(message, response.text)
    except:
        bot.reply_to(message, "فيه ضغط. جرب بعد دقيقة")

@bot.message_handler(content_types=['location'])
def handle_location(message):
    lat = message.location.latitude
    lon = message.location.longitude
    bot.reply_to(message, f"تم حفظ الموقع ✅\nhttps://maps.google.com/?q={lat},{lon}\nالحين ارسل صورة المخالفة")
